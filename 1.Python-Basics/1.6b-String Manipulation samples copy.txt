def caesar(s,k):
    res=""
    for c in s:
        if c.isalpha():
            base=65 if c.isupper() else 97
            res+=chr((ord(c)-base+k)%26+base)
        else:
            res+=c
    return res

3Ô∏è‚É£1Ô∏è‚É£ Longest Substring Without Repeating
def longest_unique(s):
    seen={}
    l=max_len=0
    for r,c in enumerate(s):
        if c in seen and seen[c]>=l:
            l=seen[c]+1
        seen[c]=r
        max_len=max(max_len,r-l+1)
    return max_len


Time: O(n)

3Ô∏è‚É£2Ô∏è‚É£ String Compression
def compress(s):
    res,count="",1
    for i in range(1,len(s)):
        if s[i]==s[i-1]:
            count+=1
        else:
            res+=s[i-1]+str(count)
            count=1
    return res+s[-1]+str(count)

3Ô∏è‚É£5Ô∏è‚É£ Longest Palindromic Substring (Center Expand)
def longest_pal(s):
    res=""
    for i in range(len(s)):
        for l,r in [(i,i),(i,i+1)]:
            while l>=0 and r<len(s) and s[l]==s[r]:
                if r-l+1>len(res):
                    res=s[l:r+1]
                l-=1;r+=1
    return res

from collections import defaultdict

def group(words):
    d=defaultdict(list)
    for w in words:
        d["".join(sorted(w))].append(w)
    return list(d.values())

3Ô∏è‚É£7Ô∏è‚É£ Minimum Window Substring (Classic Hard)

(Sliding window ‚Äî optimized)

from collections import Counter

def min_window(s,t):
    need=Counter(t)
    missing=len(t)
    l=start=end=0
    for r,c in enumerate(s,1):
        if need[c]>0:
            missing-=1
        need[c]-=1
        if missing==0:
            while l<r and need[s[l]]<0:
                need[s[l]]+=1;l+=1
            if end==0 or r-l<end-start:
                start,end=l,r
            need[s[l]]+=1;missing+=1;l+=1
    return s[start:end]

3Ô∏è‚É£8Ô∏è‚É£ Check Valid Parentheses
def valid(s):
    stack=[]
    d={")":"(","]":"[","}":"{"}
    for c in s:
        if c in d.values():
            stack.append(c)
        elif c in d:
            if not stack or stack.pop()!=d[c]:
                return False
    return not stack

4Ô∏è‚É£0Ô∏è‚É£ Longest Common Prefix
def lcp(strs):
    prefix=strs[0]
    for s in strs[1:]:
        while not s.startswith(prefix):
            prefix=prefix[:-1]
    return prefix
